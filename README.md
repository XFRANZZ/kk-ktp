```
$ pkg install php
$ pkg install git
$ git clone https://github.com/XFRANZZ/kk-ktp
$ cd kkktp

```
$ sudo apt-get update && sudo apt-get upgrade -y
$ sudo apt install php
$ sudo apt unstall git
$ git clone https://github.com/XFRANZZ/kk-ktp
$ cd kkktp
